% f c a m p
% f c a b m
% f b
% c b p
% f c a m p

\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[sibling distance=6em,
  every node/.style = {shape=rectangle, rounded corners,
    draw, align=center,
    top color=white, bottom color=blue!20}]]
  \node {root}
    child { node [color=red] {f:3}
      child { node {c:2}
        child { node {a:2}
           child { node {m:1} child { node {p:1} }  }
           child { node {b:1} child { node {m:1} }  }
         }}
      child { node [color=red] {b:1} }}
   ;
\end{tikzpicture}
\end{document}
